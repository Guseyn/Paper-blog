create database NP CHARACTER SET utf8 COLLATE utf8_general_ci;
use NP;

CREATE TABLE ussd
(
id INT NOT NULL AUTO_INCREMENT,       
name CHAR(50) NOT NULL,
fname CHAR(50) NOT NULL,
login CHAR(30) NOT NULL,
email CHAR(50) NOT NULL,
password CHAR(70) NOT NULL,
pq SMALLINT(5) NOT NULL DEFAULT 0,
fq MEDIUMINT(7) NOT NULL DEFAULT 0,
aq MEDIUMINT(7) NOT NULL DEFAULT 0,
cpq MEDIUMINT(7) NOT NULL DEFAULT 0,
fvq MEDIUMINT(7) NOT NULL DEFAULT 0,
pabs LONGBLOB NOT NULL,
copies LONGBLOB NOT NULL,
faves LONGBLOB NOT NULL,
fans LONGBLOB NOT NULL,
auts LONGBLOB NOT NULL,
likes LONGBLOB NOT NULL,
dislikes LONGBLOB NOT NULL,
views LONGBLOB NOT NULL,
isable TINYINT(1) DEFAULT 1,
hash_sess_id CHAR(70) NOT NULL,
ua LONGBLOB NOT NULL,
PRIMARY KEY(id)
);

CREATE TABLE pablications
(
id INT NOT NULL AUTO_INCREMENT,
title CHAR(65) NOT NULL,
aut INT(11) NOT NULL,
likes LONGBLOB NOT NULL,
dislikes LONGBLOB NOT NULL,
views LONGBLOB NOT NULL,
rating INT(10) NOT NULL DEFAULT 0,
lq MEDIUMINT(7) NOT NULL DEFAULT 0,
dlq MEDIUMINT(7) NOT NULL DEFAULT 0,
vq MEDIUMINT(7) NOT NULL DEFAULT 0,
path CHAR(100) NOT NULL,
ava CHAR(100) NOT NULL,
date CHAR(60) NOT NULL,
corrects LONGBLOB NOT NULL,
isreadable TINYINT(1) NOT NULL DEFAULT 1,
PRIMARY KEY(id)
);

CREATE TABLE copywritings
(
id INT NOT NULL AUTO_INCREMENT,
title CHAR(65) NOT NULL,
aut INT(11) NOT NULL,
path CHAR(100) NOT NULL,
ava CHAR(100) NOT NULL,
date CHAR(60) NOT NULL,
isreadable TINYINT(1) NOT NULL DEFAULT 1,
PRIMARY KEY(id)
);
